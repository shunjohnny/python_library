🎲 配列の作成

import numpy as np # 基本的な配列作成 arr = np.array([1, 2, 3, 4, 5]) zeros = np.zeros((3, 3)) ones = np.ones((2, 4)) empty = np.empty((2, 3)) # 等間隔の数値配列 range_arr = np.arange(0, 10, 2) # 0から10まで2刻み space_arr = np.linspace(0, 1, 5) # 0から1まで5等分 # 単位行列 eye = np.eye(3) identity = np.identity(3) # ランダム配列 rand = np.random.rand(3, 3) # 一様分布 randn = np.random.randn(3, 3) # 正規分布 randint = np.random.randint(0, 10, (3, 3)) # 整数の乱数

🔄 形状操作と結合

# 形状の変更 arr = np.array([1, 2, 3, 4, 5, 6]) reshaped = arr.reshape(2, 3) flattened = reshaped.flatten() # 転置 transposed = reshaped.T # 配列の結合 arr1 = np.array([1, 2, 3]) arr2 = np.array([4, 5, 6]) concatenated = np.concatenate([arr1, arr2]) stacked_v = np.vstack([arr1, arr2]) stacked_h = np.hstack([arr1, arr2]) # 次元の追加 expanded = np.expand_dims(arr1, axis=0) squeezed = np.squeeze(expanded) # 次元の削除

✨ 演算と集計

# 基本演算 a = np.array([1, 2, 3]) b = np.array([4, 5, 6]) added = np.add(a, b) # または a + b subtracted = np.subtract(a, b) # または a - b multiplied = np.multiply(a, b) # または a * b divided = np.divide(a, b) # または a / b # 行列演算 matrix_a = np.array([[1, 2], [3, 4]]) matrix_b = np.array([[5, 6], [7, 8]]) dot_product = np.dot(matrix_a, matrix_b) cross_product = np.cross(a, b) # 集計 total = np.sum(a) mean_val = np.mean(a) max_val = np.max(a) min_val = np.min(a)

📊 統計関数

# 基本統計量 arr = np.array([1, 2, 3, 4, 5]) std_dev = np.std(arr) # 標準偏差 variance = np.var(arr) # 分散 median_val = np.median(arr) # 中央値 # パーセンタイル percentile_75 = np.percentile(arr, 75) # 相関係数 corr = np.corrcoef(arr1, arr2) # ヒストグラム hist, bins = np.histogram(arr, bins=5)

🔍 インデックス操作

# 条件に基づく選択 arr = np.array([1, 2, 3, 4, 5]) mask = arr > 3 selected = arr[mask] # インデックス取得 indices = np.where(arr > 3) # ファンシーインデックス fancy_idx = np.array([0, 2, 4]) selected = arr[fancy_idx] # ブールインデックス mask = arr % 2 == 0 even_numbers = arr[mask]

🔀 ソートと検索

# ソート arr = np.array([3, 1, 4, 1, 5, 9, 2]) sorted_arr = np.sort(arr) sorted_indices = np.argsort(arr) # 一意の値 unique_vals = np.unique(arr) # 検索 min_idx = np.argmin(arr) max_idx = np.argmax(arr) # ソート済み配列での検索 sorted_arr = np.sort(arr) insert_pos = np.searchsorted(sorted_arr, 6)

🔢 線形代数

from numpy.linalg import inv, det, eig, svd # 逆行列 matrix = np.array([[1, 2], [3, 4]]) inverse = inv(matrix) # 行列式 determinant = det(matrix) # 固有値と固有ベクトル eigenvals, eigenvecs = eig(matrix) # 特異値分解 U, s, Vh = svd(matrix) # ノルム計算 norm = np.linalg.norm(arr)
